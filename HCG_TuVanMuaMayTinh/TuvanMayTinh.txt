			FROM sysdatabases 
			WHERE  name = 'TUVANMAYTINH')

	PRINT 'db exists'

	PRINT 'dropped db'
END
GO 
--Tao moi db

ON PRIMARY(NAME = TUVANMAYTINH_DATABASE, FILENAME = 'E:\TUVANMAYTINH__DATABSE.MDF', SIZE = 5, MAXSIZE = UNLIMITED, FILEGROWTH = 2)

PRINT 'create new db successfully'
